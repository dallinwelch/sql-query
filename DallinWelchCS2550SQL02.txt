--1 
SELECT FIRST_NAME,LAST_NAME,PHONE
FROM STUDENT
WHERE TO_CHAR(registration_date, 'MM/DD/YYYY') = '02/13/2007'
ORDER BY LAST_NAME, FIRST_NAME;
--2
SELECT COURSE_NO,SECTION_ID, START_DATE_TIME
FROM SECTION
WHERE LOCATION LIKE 'L509'
ORDER BY start_date_time;
--3
SELECT COURSE_NO,SECTION_ID,START_DATE_TIME,INSTRUCTOR_ID,CAPACITY
FROM SECTION
WHERE TO_CHAR(start_date_time, 'MON-YYYY') = 'JUL-2007'
ORDER BY START_DATE_TIME, COURSE_NO;

--4
SELECT STUDENT_ID,FINAL_GRADE 
FROM ENROLLMENT
WHERE TO_CHAR(ENROLL_DATE, 'MM-YY') = '01-07'
AND FINAL_GRADE IS NOT NULL;

--5
SELECT TO_DATE('01/11/16', 'MM/DD/YY') +111 AS END_OF_SEMESTER
FROM DUAL;
--6
SELECT DISTINCT TO_CHAR(REGISTRATION_DATE, 'MM') ,  TO_CHAR(REGISTRATION_DATE, 'MONTH')
FROM STUDENT
ORDER BY TO_CHAR(registration_date, 'MM');
--7
SELECT  LAST_NAME,FIRST_NAME
FROM STUDENT
WHERE registration_date >= CREATED_DATE +3
AND ZIP = 11368;
--8
SELECT FIRST_NAME, LAST_NAME, TO_CHAR(ROUND(SYSDATE - REGISTRATION_DATE)/365, '99.99' ) AS Years
FROM STUDENT
WHERE PHONE LIKE '212%'

--9
SELECT DISTINCT TO_CHAR(start_date_time, 'HH:MI')
FROM SECTION
--10
SELECT STUDENT_ID, SECTION_ID, ENROLL_DATE
FROM ENROLLMENT
WHERE TO_CHAR(ENROLL_DATE, 'HH24:MI')= '10:19'
ORDER BY STUDENT_ID;
